1 (function( window, undefined ) {
  2 
  3     /**
  4     * The hopefully easy to setup, run everywhere user profiler.
  5     * 
  6     * <p>Sup aims to allow simple configuration in order to be reusable. 
  7     * It is technically a singleton object broken into multiple sub-objects:</p>
  8     * <ul>
  9     * <li>{@link Sup.Agents} are the autonomous elements you create to interact
 10     * with your users, you must declare them at configuration.</li>
 11     * <li>{@link Sup.Controller} takes care of data both externally (loading/saving) 
 12     * and internally (parsing/converting/distributing)</li>
 13     * <li>{@link Sup.Dispatcher} acts as a middleman between the Controller and 
 14     * the Agents actually instancing the later ones in a Factory way.</li>
 15     * <li>{@link Sup.Strategy} depends on you to be defined (State Machine, 
 16     * Behaviour Tree, etc.) and set the agents to be dispatched.</li>
 17     * </ul>
 18     * <p>The system uses a request object -owned by {@link Sup.Controller}- that
 19     * contains the user profile. It is pulled and pushed as JSON data following
 20     * read & write methods you must declare at configuration. 
 21     * </p>
 22     *  
 23     * @class the Simple User Profiler. Because users need help. 
 24     */
 25     Sup =  {};
 26 
 27 
 28     /** 
 29      * This is the principal method you'll use : it sets things up while keeping the internals away 
 30      * @param params your configuration object 
 31      * @return self for fluent interface Sup.configure(...).run()
 32      **/
 33     Sup.configure = function ( params ) {
 34         Sup.Controller.read = params["read"] || function(){};
 35         Sup.Controller.write = params["write"] || function(){};
 36         if( params["strategy"] ){ 
 37             Sup.Strategy._fn = params["strategy"];
 38         }
 39         if( params["agents"] ){ 
 40             Sup.Agents._list = params["agents"];
 41         }
 42         return Sup;
 43     };
 44     
 45     /** Once configured, let Sup run.
 46      **/
 47     Sup.run = function(){
 48         Sup.Controller.load();
 49         Sup.Dispatcher._list = Sup.Strategy.run();
 50         Sup.Dispatcher.run();
 51     };
 52         
 53     /** When in doubt, use this method to have Sup return it's own debugging info
 54      * @usage in console : Sup.debug()
 55      **/
 56     Sup.debug = function(){
 57         for( var i in Sup){
 58             if( typeof( Sup[i] ) == "object"){
 59                 Sup[i]._debug();
 60             }
 61         }    
 62     }
 63     
 64 
 65     // Agents
 66    
 67    Sup.Agents = { /** @lends Sup.Agents.prototype */
 68             /** 
 69             * The Agents object acts as a pool of potential objects you configure
 70             * @class Objects pool
 71             * @name Sup.Agents 
 72             **/
 73         };
 74     Sup.Agents._debug = function(){}
 75     
 76     /**
 77     * Provides a standard interface for all your agents
 78     * @class Abstract 
 79     */
 80     Sup.Agents.Abstract = function(name){ /** @lends Sup.Agents.Abstract.prototype */
 81         
 82         // The name of your action
 83         this._name = name || "unknown";
 84 
 85         /** Reads the request to set all properties on your agent */
 86         this.init = function(){
 87             var request = Sup.Controller.request();
 88             if( undefined != request["agents"][this._name]){
 89                 for( var i in request["agents"][this._name] ){
 90                     this[i] = request["agents"][this._name][i]
 91                 }
 92             }
 93         }
 94         /** Serializes your agent properties and pushes it to controller subscribing a save agent */ 
 95         this.update = function(do_save){
 96             var serializable = {};
 97             for( var i in this){
 98     //            if( typeof( this [i] ) == "function" ){
 99     //                continue;
100     //            }
101                 if( typeof( i ) == "string" ){
102                     if ( i[0] == "_"){
103                         continue;
104                     }
105                 }
106                 serializable[i] = this [i];
107             }
108             Sup.Controller.requestPart(this._name,serializable);
109             if( do_save ){
110                 Sup.Controller.save();
111             }
112         }
113     }
114 
115 
116     // Strategy
117         
118     Sup.Strategy = {
119             /** 
120             * Strategy offers you a way to deal freely with the logic
121             * of the profiler, implementing any kind of decision algorithm
122             * that will be called by {@link Sup.Controller} at runtime.
123             * It returns the {@link Sup.Agents} instanced by {@link Sup.Dispatcher}
124             * @class Strategy / Algorithmic decisional
125             * @name Sup.Strategy 
126             **/
127         };
128         
129     /** Method you configure that implements dispatcher "AI" */
130     Sup.Strategy._fn = function(){}
131     
132     /** AI function executer, performs a selection of agents */
133     Sup.Strategy.run = function(){
134         return this._fn(Sup.Controller.request());
135     }
136     
137     Sup.Strategy._debug = function(){}
138 
139     // Dispatcher
140 
141    Sup.Dispatcher = { /** @lends Sup.Dispatcher.prototype */
142             /** 
143             * Dispatcher is a mechanical part as it receives a list generated 
144             * by {@link Sup.Strategy} to instance the {@link Sup.Agents}
145             * @class Multi agents handler / Factory  
146             * @name Sup.Dispatcher 
147             **/
148         };
149     /** Factories your Agents and records them */
150     Sup.Dispatcher.run = function(){
151         this._dispatched = {};
152         for( var i in this._list ){
153             if( "function" == typeof( this._list[i] ) ){
154                var agent = new Sup.Agents.Abstract(i);
155                agent.init(Sup.Controller.request())
156                agent.run = this._list[i]
157                agent.run()
158                this._dispatched[i] = agent;
159             }
160         }
161     }
162 
163     Sup.Dispatcher._debug = function(){}
164 
165 
166     // Controller 
167     
168     Sup.Controller = { /** @lends Sup.Controller.prototype */
169             /** 
170             * Contains the request object, loads and saves it for you. The 
171             * "request" is 
172             * @class IO Controller
173             * 
174             * @name Sup.Controller 
175             **/
176             "_request" : {},
177             
178             /** User rewritable function */
179             "write" : function(){}
180         };
181         
182 
183     /** Loads your user data, parses it and sets it as the request */
184     Sup.Controller.load = function(){
185         Sup.Controller._request = JSON.parse( Sup.Controller.read() );
186         if( undefined == this._request["agents"]){
187             this._request["agents"] = {};
188         }
189     }
190     
191     /** User rewritten function */
192     Sup.Controller.read = function(){}
193             
194     /** User rewritten function */
195     Sup.Controller.write = function(){}
196             
197     /** Encodes the request and puts it where you asked it to */
198     Sup.Controller.save = function(){
199         Sup.Controller.write( Sup.Controller.request() );
200     }
201     
202     /** Your basic getter for the request object */
203     Sup.Controller.request = function(){
204         return Sup.Controller._request;
205     };
206     
207     /** Acts as a getter setter for agents parts of the request */
208     Sup.Controller.requestPart = function(part,content){
209         // getter
210         if( undefined == content){
211             return this._request["agents"][part];
212         }
213         // setter
214         this._request["agents"][part] = content;
215     }
216 
217     Sup.Controller._debug = function(){
218         if( "function" != typeof( console.log )){
219             alert("No console available.");
220         }
221         console.log("request",this._request);
222         for( var i in this._request.agents){
223             console.log( "  request agent: "+i);
224             for( var ii in this._request.agents[i]){
225                 if( "function" != typeof( this._request.agents[i][ii] ) ){
226                     console.log( "    "+ii+" ",this._request.agents[i][ii]);
227                 }
228             }
229         }
230     }
231     
232 
233   window.Sup = Sup;

235 })( window );
